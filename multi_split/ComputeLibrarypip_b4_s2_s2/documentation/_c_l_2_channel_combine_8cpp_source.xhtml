47 inline void validate_configuration(const TensorShape &shape, Format format)
48 {
49  const int num_planes = num_planes_from_format(format);
50 
51  // Create tensors
52  CLMultiImage dst = create_multi_image<CLMultiImage>(shape, format);
53  std::vector<CLTensor> ref_src = create_tensor_planes<CLTensor>(shape, format);
54 
55  // Create and configure function
56  CLChannelCombine channel_combine;
57 
58  if(num_planes == 1)
59  {
60  const CLTensor *tensor_extra = ((Format::RGBA8888 == format) ? &ref_src[3] : nullptr);
61 
62  channel_combine.configure(&ref_src[0], &ref_src[1], &ref_src[2], tensor_extra, dst.cl_plane(0));
63  }
64  else
65  {
66  channel_combine.configure(&ref_src[0], &ref_src[1], &ref_src[2], &dst);
67  }
68 }
